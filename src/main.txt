#include <Arduino.h>
#include <WiFi.h>
#include <ArduinoWebsockets.h>
#include <ArduinoJson.h>
#include <U8g2lib.h>
#include <Wire.h>
#include "config.h"

using namespace websockets;

// ============================================
// Global Objects
// ============================================
U8G2_SSD1306_128X64_NONAME_F_HW_I2C display(
  U8G2_R0,
  U8X8_PIN_NONE
);

WebsocketsClient wsClient;

// ============================================
// State Variables
// ============================================
enum ConnectionState {
  DISCONNECTED,
  CONNECTING,
  CONNECTED,
  REGISTERED
};

ConnectionState connectionState = DISCONNECTED;

unsigned long lastHeartbeat = 0;
unsigned long lastReconnectAttempt = 0;

String statusMessage = "Starting...";
bool displayNeedsUpdate = true;

// ============================================
// Function Declarations
// ============================================
void setupWiFi();
void setupDisplay();
void setupWebSocket();
void connectWebSocket();
void sendHeartbeat();
void registerDevice();
void handleWebSocketMessage(WebsocketsMessage message);
void handleWebSocketEvent(WebsocketsEvent event, String data);
void updateDisplay();
void displayStatus(const char* line1, const char* line2 = "", const char* line3 = "");

// ============================================
// Setup
// ============================================
void setup() {
  #if DEBUG_SERIAL
  Serial.begin(SERIAL_BAUD);
  delay(1000);
  Serial.println("\n\n=================================");
  Serial.println("ESP32 Smart Panel Starting...");
  Serial.println("=================================");
  #endif

  // Initialize display first
  setupDisplay();
  displayStatus("Initializing...", "Please wait");

  // Setup WiFi
  setupWiFi();

  // Setup WebSocket
  setupWebSocket();

  // Initial connection
  connectWebSocket();

  Serial.println("✅ Setup complete!");
}

// ============================================
// Main Loop
// ============================================
void loop() {
  // Handle WebSocket
  if (wsClient.available()) {
    wsClient.poll();
  }

  // Auto-reconnect if disconnected
  if (connectionState == DISCONNECTED) {
    unsigned long now = millis();
    if (now - lastReconnectAttempt > RECONNECT_DELAY) {
      lastReconnectAttempt = now;
      connectWebSocket();
    }
  }

  // Send heartbeat
  if (connectionState == REGISTERED) {
    unsigned long now = millis();
    if (now - lastHeartbeat > HEARTBEAT_INTERVAL) {
      lastHeartbeat = now;
      sendHeartbeat();
    }
  }

  // Update display if needed
  if (displayNeedsUpdate) {
    updateDisplay();
    displayNeedsUpdate = false;
  }

  delay(10);
}

// ============================================
// WiFi Setup
// ============================================
void setupWiFi() {
  Serial.println("\n📡 Connecting to WiFi...");
  displayStatus("WiFi", "Connecting...");

  WiFi.mode(WIFI_STA);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);

  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 30) {
    delay(500);
    Serial.print(".");
    attempts++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\n✅ WiFi Connected!");
    Serial.print("📍 IP Address: ");
    Serial.println(WiFi.localIP());
    
    displayStatus("WiFi Connected", WiFi.localIP().toString().c_str());
    delay(2000);
  } else {
    Serial.println("\n❌ WiFi connection failed!");
    displayStatus("WiFi Failed", "Check config");
    delay(5000);
  }
}

// ============================================
// Display Setup
// ============================================
void setupDisplay() {
  Wire.begin(OLED_SDA, OLED_SCL);
  
  if (!display.begin()) {
    Serial.println("❌ OLED initialization failed!");
    return;
  }

  display.clearBuffer();
  display.setFont(u8g2_font_ncenB08_tr);
  display.drawStr(0, 20, "Smart Panel");
  display.drawStr(0, 35, "Initializing...");
  display.sendBuffer();
  
  Serial.println("✅ Display initialized");
}

// ============================================
// WebSocket Setup
// ============================================
void setupWebSocket() {
  // Setup message handler
  wsClient.onMessage(handleWebSocketMessage);
  
  // Setup event handler
  wsClient.onEvent(handleWebSocketEvent);

  Serial.println("✅ WebSocket handlers configured");
}

// ============================================
// WebSocket Connection
// ============================================
void connectWebSocket() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("⚠️ WiFi not connected, skipping WebSocket connection");
    return;
  }

  Serial.println("\n🔌 Connecting to WebSocket server...");
  displayStatus("WebSocket", "Connecting...");
  connectionState = CONNECTING;

  String wsUrl = "ws://" + String(WS_SERVER_HOST) + ":" + 
                 String(WS_SERVER_PORT) + String(WS_SERVER_PATH);
  
  Serial.print("📍 URL: ");
  Serial.println(wsUrl);

  bool connected = wsClient.connect(wsUrl);

  if (connected) {
    Serial.println("✅ WebSocket connected!");
    connectionState = CONNECTED;
    
    // Register device
    delay(500);
    registerDevice();
  } else {
    Serial.println("❌ WebSocket connection failed!");
    connectionState = DISCONNECTED;
    displayStatus("WS Failed", "Retrying...");
  }
}

// ============================================
// Device Registration
// ============================================
void registerDevice() {
  Serial.println("📝 Registering device...");

  JsonDocument doc;
  doc["type"] = "register";
  doc["deviceId"] = DEVICE_ID;
  doc["deviceType"] = "esp32";
  doc["data"]["name"] = DEVICE_NAME;
  doc["data"]["ip"] = WiFi.localIP().toString();
  doc["data"]["mac"] = WiFi.macAddress();
  doc["timestamp"] = millis();

  String message;
  serializeJson(doc, message);

  wsClient.send(message);
  Serial.println("✅ Registration sent");
}

// ============================================
// Heartbeat
// ============================================
void sendHeartbeat() {
  JsonDocument doc;
  doc["type"] = "heartbeat";
  doc["deviceId"] = DEVICE_ID;
  doc["timestamp"] = millis();

  String message;
  serializeJson(doc, message);

  wsClient.send(message);
  
  #if DEBUG_SERIAL
  Serial.println("💓 Heartbeat sent");
  #endif
}

// ============================================
// WebSocket Message Handler
// ============================================
void handleWebSocketMessage(WebsocketsMessage message) {
  Serial.println("\n📨 Message received:");
  Serial.println(message.data());

  JsonDocument doc;
  DeserializationError error = deserializeJson(doc, message.data());

  if (error) {
    Serial.print("❌ JSON parse error: ");
    Serial.println(error.c_str());
    return;
  }

  const char* type = doc["type"];
  Serial.print("📋 Type: ");
  Serial.println(type);

  // Handle different message types
  if (strcmp(type, "status") == 0) {
    const char* status = doc["data"]["status"];
    
    if (strcmp(status, "registered") == 0) {
      Serial.println("✅ Device registered successfully!");
      connectionState = REGISTERED;
      statusMessage = "Connected";
      displayNeedsUpdate = true;
      lastHeartbeat = millis();
    }
  }
  else if (strcmp(type, "command") == 0) {
    Serial.println("🎯 Command received");
    
    // Extract command data
    const char* command = doc["data"]["command"];
    Serial.print("   Command: ");
    Serial.println(command);

    // Handle commands
    if (strcmp(command, "display_message") == 0) {
      const char* text = doc["data"]["text"];
      statusMessage = String(text);
      displayNeedsUpdate = true;
      
      // Send acknowledgment
      JsonDocument ackDoc;
      ackDoc["type"] = "status";
      ackDoc["deviceId"] = DEVICE_ID;
      ackDoc["data"]["status"] = "command_executed";
      ackDoc["data"]["command"] = command;
      
      String ackMessage;
      serializeJson(ackDoc, ackMessage);
      wsClient.send(ackMessage);
    }
  }
  else if (strcmp(type, "notification") == 0) {
    Serial.println("🔔 Notification received");
    
    const char* event = doc["data"]["event"];
    Serial.print("   Event: ");
    Serial.println(event);
    
    // Show notification on display
    statusMessage = "Event: " + String(event);
    displayNeedsUpdate = true;
  }
}

// ============================================
// WebSocket Event Handler
// ============================================
void handleWebSocketEvent(WebsocketsEvent event, String data) {
  switch(event) {
    case WebsocketsEvent::ConnectionOpened:
      Serial.println("✅ WebSocket connection opened");
      break;
      
    case WebsocketsEvent::ConnectionClosed:
      Serial.println("❌ WebSocket connection closed");
      connectionState = DISCONNECTED;
      statusMessage = "Disconnected";
      displayNeedsUpdate = true;
      break;
      
    case WebsocketsEvent::GotPing:
      Serial.println("💓 Got ping");
      break;
      
    case WebsocketsEvent::GotPong:
      Serial.println("💓 Got pong");
      break;
  }
}

// ============================================
// Display Update
// ============================================
void updateDisplay() {
  display.clearBuffer();
  
  // Draw header
  display.setFont(u8g2_font_ncenB08_tr);
  display.drawStr(0, 10, DEVICE_NAME);
  display.drawLine(0, 12, 128, 12);
  
  // Draw status
  display.setFont(u8g2_font_6x10_tr);
  
  // Connection status
  const char* connStatus = "";
  switch(connectionState) {
    case DISCONNECTED:
      connStatus = "[ Offline ]";
      break;
    case CONNECTING:
      connStatus = "[ Connecting... ]";
      break;
    case CONNECTED:
      connStatus = "[ Connected ]";
      break;
    case REGISTERED:
      connStatus = "[ Active ]";
      break;
  }
  display.drawStr(0, 25, connStatus);
  
  // Status message
  display.drawStr(0, 40, statusMessage.c_str());
  
  // Footer - IP address
  display.setFont(u8g2_font_5x7_tr);
  String ip = WiFi.localIP().toString();
  display.drawStr(0, 60, ip.c_str());
  
  display.sendBuffer();
}

// ============================================
// Display Status Helper
// ============================================
void displayStatus(const char* line1, const char* line2, const char* line3) {
  display.clearBuffer();
  display.setFont(u8g2_font_ncenB08_tr);
  
  display.drawStr(0, 15, line1);
  
  if (strlen(line2) > 0) {
    display.setFont(u8g2_font_6x10_tr);
    display.drawStr(0, 35, line2);
  }
  
  if (strlen(line3) > 0) {
    display.drawStr(0, 50, line3);
  }
  
  display.sendBuffer();
}
